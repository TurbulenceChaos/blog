\nabla \cdot \mathbf{u} = 0 \tag{1} \\\\
\frac{\partial \mathbf{u}}{\partial t} + (\mathbf{u} \cdot \nabla)\mathbf{u} =
-\frac{1}{\rho} \nabla p + \nu \nabla^{2} \mathbf{u} \tag{2}
\end{gather}
